// Program ini akan mengkonversi cm ke m dan km

// DEKLARASI VARIABLE
var input_cm: long long
typedef struct {
  long long cm;
  long long m;
  long long km;
} Hasil
var res: Hasil

// deklarasi algoritma

// meminta masukan kepada user
input_cm <- read("Masukkan nilai panjang dalam satuan cm: ");
printf("%lli cm = ", input_cm);

// proses konversi
while input_cm >= 100000 do
  input_cm -= 100000;
  res.km += 1;
end while 

while input_cm >= 100 do
  input_cm -= 100;
  res.m += 1;
end while

res.cm <- input_cm;

// menampilkan hasil konversi
write("%lli km + ", res.km);
write("%lli m + ", res.m);
write("%lli cm", res.cm);
